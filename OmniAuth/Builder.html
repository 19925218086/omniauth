def provider(klass, *args, &block)
  if klass.is_a?(Class)
    middleware = klass
  else
    middleware = OmniAuth::Strategies.const_get("#{OmniAuth::Utils.camelize(klass.to_s)}")
  end
  
  use middleware, *args, &block
end
