def self.discover(uri)
  discovered = self.default_discover(uri)
  
  if discovered.last.empty?
    info = discover_google_apps(uri) 
    return info if info
  end
  
  return discovered
rescue OpenID::DiscoveryFailure => e
  info = discover_google_apps(uri)
  
  if info.nil?
    raise e
  else
    return info
  end
end
